% ---------------------------------------------------------------
% Correção de entradas duplicadas no Sumário para \chapter* :
% repUERJ's \@schapter já adiciona entrada com \bf\uppercase ao TOC.
% Pandoc também adiciona \addcontentsline após \chapter*, gerando
% duplicata em caixa normal. Suprimimos a chamada extra do Pandoc
% com uma flag que é ativada após \@schapter processar a sua entrada.
% ---------------------------------------------------------------
\makeatletter
\newif\if@uerj@suppress@nexttoc
\@uerj@suppress@nexttocfalse

\let\@uerj@orig@schapter\@schapter
\def\@schapter#1{%
  \@uerj@orig@schapter{#1}%
  \@uerj@suppress@nexttoctrue
}

\let\@uerj@orig@addcontentsline\addcontentsline
\def\addcontentsline#1#2#3{%
  \if@uerj@suppress@nexttoc
    \@uerj@suppress@nexttocfalse
  \else
    \@uerj@orig@addcontentsline{#1}{#2}{#3}%
  \fi
}
\makeatother
